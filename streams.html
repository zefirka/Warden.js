Streams

From DOM elements

Warden(document);
var clicks = document.stream('keydown');

From custom callback

var clicks = Warden.Stream(function(emit){
  document.addEventListener('keydown', emit);
});

Processing

Map

var chars = keydowns
  .map('.keyCode')
  .map(String.fromCharCode)

Filter

var letters = chars
  .filter(function(ch){
    return "0123456789".indexOf(ch) == -1;
  });

Reduce

var inputs = chars
      .reduce('', function(res, ch){
        return res.concat(ch);
      });
