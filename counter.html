Counter

Counter which merges two streams

+ 1 - 1 0

Implementation

Streams

var add = plus.stream('click').map(1),
    sub = minus.stream('click').map(-1),
    results = add.merge(sub).reduce(0, function(a, b){
      return a+b;
    });
