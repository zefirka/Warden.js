Tutorials and examples

Vanilla Warden 
var a = 10;
var b = 20;
var c = a + b;
console.log(c); 

// -> 30;

a = 20;
console.log(c);

// -> 30; 

var a = Warden(10);
var b = Warden(20);
var c = Warden.Formula([a,b], function(x,y){
	return x + y;
});

console.log(c.value); 
// -> 30

a.value = 20;
console.log(c.value); 
// -> 40  wow! It's updates

var timer;
button.addEventListener('click', function(){
	var self = this;
	clearTimeout(timer);
	timer = setTimeout(function(){
		send(self.value);
	}, 500)
});

Warden(button)
	.stream('click')
	.debounce(500)
	.map('@value')
	.listen(send);
